392 const char* getAccessFlags(u2 accessFlags) {
393  const char* flags[] = { "public ", "final ", "super ", "interface ", "abstract ", "private ", "protected ", "static ", "volatile ",
394  "transient " };
395  uint16_t masks[] = { 0x0001, 0x0010, 0x0020, 0x0200, 0x0400, 0x0002, 0x0004, 0x0008, 0x0040, 0x0080 };
396 
397  stringstream ss;
398  for (uint8_t i = 0; i < 10; i++) {
399  if ((accessFlags & masks[i]) != 0) {
400  ss << flags[i];
401  }
402  }
403 
404  return Utils::converterStreamToCString(ss);
405 }

759 void printArquivoByteCode(Code_attribute codeAttribute, cp_info *constantPool, uint8_t indentation) {
760  u4 code_length = codeAttribute.code_length;
761  u1 *code = codeAttribute.code;
762 
763  uint32_t lineNumber = 1;
764  u4 i = 0;
765  while (i < code_length) {
766  Utils::imprintTabs(out, indentation);
767  fprintf(out, "%d\t%d\t%s", lineNumber++, i, vetorNomeFuncao[code[i]]);
768 
769  if (code[i] <= 0x0f || (code[i] >= 0x1a && code[i] <= 0x35) || (code[i] >= 0x3b && code[i] <= 0x83)
770  || (code[i] >= 0x85 && code[i] <= 0x98) || (code[i] >= 0xac && code[i] <= 0xb1) || code[i] == 0xbe || code[i] == 0xbf
771  || code[i] == 0xc2 || code[i] == 0xc3) {
772  fprintf(out, "\n");
773  i += 1;
774  } else if (code[i] == 0x12) { // usa CP
775  fprintf(out, " #%d <%s>\n", code[i + 1], getFormattedConstant(constantPool, code[i + 1]));
776  i += 2;
777  } else if (code[i] == 0xbc) { // newarray
778  u1 atype = code[i + 1];
779  const char* types[] = { "boolean", "char", "float", "double", "byte", "short", "int", "long" };
780  fprintf(out, " %d (%s)\n", atype, types[atype - 4]);
781  i += 2;
782  } else if (code[i] == 0x10 || (code[i] >= 0x15 && code[i] <= 0x19) || (code[i] >= 0x36 && code[i] <= 0x3a) || code[i] == 0xa9) {
783  fprintf(out, " %d\n", code[i + 1]);
784  i += 2;
785  } else if (code[i] == 0x11) { // sipush
786  int16_t number = (code[i + 1] << 8) | code[i + 2];
787  fprintf(out, " %d\n", number);
788  i += 3;
789  } else if (code[i] == 0x84) { // iinc
790  fprintf(out, " %d by %d\n", code[i + 1], code[i + 2]);
791  i += 3;
792  } else if (code[i] == 0x13 || code[i] == 0x14 || (code[i] >= 0xb2 && code[i] <= 0xb8) || code[i] == 0xbb || code[i] == 0xbd
793  || code[i] == 0xc0 || code[i] == 0xc1) { // usa CP
794  u2 number = (code[i + 1] << 8) | code[i + 2];
795  fprintf(out, " #%d <%s>\n", number, getFormattedConstant(constantPool, number));
796  i += 3;
797  } else if ((code[i] >= 0x99 && code[i] <= 0xa8) || code[i] == 0xc6 || code[i] == 0xc7) {
798  int16_t number = (code[i + 1] << 8) | code[i + 2];
799  fprintf(out, " %d (%+d)\n", i + number, number);
800  i += 3;
801  } else if (code[i] == 0xc5) { // multianewarray - usa CP
802  u2 number = (code[i + 1] << 8) | code[i + 2];
803  fprintf(out, " #%d <%s> dim %d\n", number, getFormattedConstant(constantPool, number), code[i + 3]);
804  i += 4;
805  } else if (code[i] == 0xb9) { // invokeinterface - usa CP
806  u2 number = (code[i + 1] << 8) | code[i + 2];
807  fprintf(out, " #%d <%s> count %d\n", number, getFormattedConstant(constantPool, number), code[i + 3]);
808  assert(code[i + 4] == 0);
809  i += 5;
810  } else if (code[i] == 0xc8 || code[i] == 0xc9) { // goto_w e jsr_w
811  int32_t number = (code[i + 1] << 24) | (code[i + 2] << 16) | (code[i + 3] << 8) | code[i + 4];
812  fprintf(out, " %d (%+d)\n", i + number, number);
813  i += 5;
814  } else if (code[i] == 0xc4) { // wide
815  u2 indexbyte = (code[i + 2] << 8) | code[i + 3];
816 
817  fprintf(out, "\n");
818  Utils::imprintTabs(out, indentation);
819  fprintf(out, "%d\t%d\t%s %d", lineNumber++, i + 1, vetorNomeFuncao[code[i + 1]], indexbyte);
820 
821  if (code[i + 1] == 0x84) { // format 2 (iinc)
822  int16_t constbyte = (code[i + 4] << 8) | code[i + 5];
823  fprintf(out, " by %d", constbyte);
824  i += 6;
825  } else {
826  i += 4;
827  }
828 
829  fprintf(out, "\n");
830  } else if (code[i] == 0xaa) { // tableswitch
831  u1 padding = (i + 1) % 4;
