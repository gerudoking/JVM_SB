94 method_info* Frame::obterMethodNamed(StaticClass *classRuntime, string name, string descriptor) {
95  MethodArea &methodArea = MethodArea::getInstance();
96 
97  StaticClass *currClass = classRuntime;
98  method_info *method;
99 
100  while (currClass != NULL) {
101  ClassFile *classFile = currClass->getClassFile();
102 
103  for (int i = 0; i < classFile->methods_count; i++) {
104  method = &(classFile->methods[i]);
105  string methodName = getFormattedConstant(classFile->constant_pool, method->name_index);
106  string methodDesc = getFormattedConstant(classFile->constant_pool, method->descriptor_index);
107 
108  if (methodName == name && methodDesc == descriptor) {
109  _classRuntime = currClass;
110  return method;
111  }
112  }
113 
114  // procurando o método nas super classes.
115  if (classFile->super_class == 0) {
116  currClass = NULL;
117  } else {
118  string superClassName = getFormattedConstant(classFile->constant_pool, classFile->super_class);
119  currClass = methodArea.carregarClassNamed(superClassName);
120  }
121  }
122 
123  return NULL;
124 }
